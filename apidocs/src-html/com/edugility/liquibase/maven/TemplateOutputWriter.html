028    package com.edugility.liquibase.maven;
029    
030    import java.io.IOException;
031    import java.io.Writer;
032    
033    import org.mvel2.templates.util.TemplateOutputStream;
034    
035    import org.mvel2.templates.util.io.StandardOutputStream; // for javadoc only
036    
037    /**
038     * A {@link TemplateOutputStream} that delegates to a {@link Writer}.
039     *
040     * <h3>Design Notes</h3>
041     *
042     * <p>The <a href="http://mvel.codehaus.org/">MVEL</a> project's
043     * {@link TemplateOutputStream} class is really a {@code char}-based
044     * construct, and hence has semantics very close to the {@link Writer}
045     * class, despite the fact that it sounds by convention like it should
046     * work with {@code byte} streams.</p>
047     *
048     * @author <a href="http://about.me/lairdnelson"
049     * target="_parent">Laird Nelson</a>
050     *
051     * @see TemplateOutputStream
052     *
053     * @see Writer
054     */
055    public class TemplateOutputWriter implements TemplateOutputStream {
056    
057      /**
058       * The {@link Writer} instance to which all work is delegated.
059       *
060       * <p>This field is never {@code null}.</p>
061       *
062       * @see #TemplateOutputWriter(Writer)
063       */
064      private final Writer delegate;
065    
066      /**
067       * Creates a new {@link TemplateOutputWriter}.
068       *
069       * @param delegate the {@link Writer} to which all operations will
070       * be delegated; must not be {@code null}.  This {@link Writer} is
071       * never {@linkplain Writer#close() closed} by any of this {@link
072       * TemplateOutputWriter}'s operations.
073       *
074       * @exception IllegalArgumentException if {@code delegate} is {@code
075       * null}
076       */
077      public TemplateOutputWriter(final Writer delegate) {
078        super();
079        if (delegate == null) {
080          throw new IllegalArgumentException("delegate", new NullPointerException("delegate"));
081        }
082        this.delegate = delegate;
083      }
084    
085      /**
086       * {@linkplain Writer#append(CharSequence) Appends} the supplied
087       * {@link CharSequence} to the {@link Writer} supplied {@linkplain
088       * #TemplateOutputWriter(Writer) at construction time} and returns
089       * this {@link TemplateOutputWriter}.
090       *
091       * <p>This method never returns {@code null}.</p>
092       *
093       * @param c the {@link CharSequence} in question; may be {@code
094       * null} in which case "{@code null}" is appended instead
095       *
096       * @return this {@link TemplateOutputWriter}; never {@code null}
097       *
098       * @exception RuntimeException if any underlying {@link IOException}
099       * is thrown in keeping with the somewhat questionable design of
100       * {@link TemplateOutputStream}; examine its {@linkplain
101       * Throwable#getCause() root cause} for the causing {@link
102       * IOException}
103       *
104       * @see Writer#append(CharSequence)
105       */
106      @Override
107      public TemplateOutputStream append(final CharSequence c) {
108        try {
109          this.delegate.append(c);
110        } catch (final IOException e) {
111          // This choice of RuntimeException and its message were made to
112          // line up with org.mvel2.templates.util.io.StandardOutputStream
113          // which handles errors in exactly this manner for better or for
114          // worse.
115          throw new RuntimeException("failed to write to stream", e);
116        }
117        return this;
118      }
119    
120      /**
121       * {@linkplain Writer#write(char[]) Writes} the supplied {@code
122       * char} array to the {@link Writer} supplied {@linkplain
123       * #TemplateOutputWriter(Writer) at construction time} and returns
124       * this {@link TemplateOutputWriter}.
125       *
126       * <p>This method never returns {@code null}.</p>
127       *
128       * @param c the {@code char} array in question; per the
129       * documentation for the {@link Writer#write(char[])} method,
130       * behavior is unspecified if this parameter is {@code null}
131       *
132       * @exception RuntimeException if any underlying {@link IOException}
133       * is thrown in keeping with the somewhat questionable design of
134       * {@link TemplateOutputStream}; examine its {@linkplain
135       * Throwable#getCause() root cause} for the causing {@link
136       * IOException}
137       *
138       * @see Writer#append(CharSequence)
139       */
140      @Override
141      public TemplateOutputStream append(final char[] c) {
142        try {
143          this.delegate.write(c);
144        } catch (final IOException e) {
145          // This choice of RuntimeException and its message were made to
146          // line up with org.mvel2.templates.util.io.StandardOutputStream
147          // which handles errors in exactly this manner for better or for
148          // worse.
149          throw new RuntimeException("failed to write to stream", e);
150        }
151        return this;
152      }
153    
154      /**
155       * Returns {@code null} when invoked, following the undocumented
156       * practice and convention of the {@link StandardOutputStream}
157       * class.
158       *
159       * @return {@code null} when invoked
160       *
161       * @see StandardOutputStream#toString()
162       */
163      @Override
164      public String toString() {
165        // The purpose of the toString() method of an arbitrary
166        // TemplateOutpuStream is not documented anywhere in MVEL so we
167        // emulate the behavior of
168        // org.mvel2.templates.util.io.StandardOutputStream, which returns
169        // null from this method.  The whole thing smells like some MVEL
170        // code somewhere might call this method to see if this particular
171        // TemplateOutputStream is capable of returning all the stuff that
172        // was written to it as a String, which this implementation most
173        // certainly is not.
174        return null;
175      }
176    
177    }
